# Interfaces

## IBlockType

```tsx
interface IBlockType {
  name: string
  label: string
  getDefaultProps: () => object
  hideFromAddMenu?: boolean
  sideEditProps?: Array<ISideEditProp | ISideGroup>
  repeaterItems?: IRepeaterItem[]
}
```

## IRepeaterItem

```tsx
interface IRepeaterItem {
  name: string
  itemType: string
  itemLabel?: string
  min?: number
  max?: number
}
```

interface IContentBlock {
